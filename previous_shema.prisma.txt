generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql" // or mysql, sqlserver, sqlite, mongodb
}

model settings {
  id            Int      @id @default(autoincrement())
  template_name String   @unique
  created_at    DateTime @default(now())
  active        Boolean  @default(false)
}

model User {
  id         Int      @id @default(autoincrement())
  username   String   @unique
  email      String   @unique
  created_at DateTime @default(now())
  active     Boolean  @default(true)

  // Relations can also be defined here
  posts Post[]

  @@map("user")
}

model Post {
  id                Int      @id @default(autoincrement())
  title             String
  content           String
  published         Boolean  @default(false)
  created_at        DateTime @default(now())
  short_description String
  image_url         String

  // Relation: each post related to a specific user from user table
  author    User @relation(fields: [author_id], references: [id]) // refrences to user model
  author_id Int // refreneces to foreign key id in user model.

  @@map("post")
}

model Product {
  id                       Int      @id @default(autoincrement())
  name                     String
  title                    String
  short_description        String
  description              String
  features                 String
  main_img_url             String
  marketing_img_url        String
  mobile_marketing_img_url String
  price                    Int
  created_at               DateTime @default(now())
  origin                   String
  url_address              String

  // Relation type with productCategories
  category ProductCategories[]

  // Relation type woth productcolors
  colors ProductColors[]

  productImages ProductImages[]

  @@map("product")
}

model Lists {
  id        Int    @id @default(autoincrement())
  list_type String

  category Category[]
  pages    Pages[]

  @@map("lists")
}

model Pages {
  id          Int    @id @default(autoincrement())
  name        String
  title       String
  description String
  url_address String
  keys        String
  list_type   Int

  list Lists @relation(fields: [list_type], references: [id])

  @@map("pages")
}

model Category {
  id          Int    @id @default(autoincrement())
  name        String
  description String
  parent_id   Int?
  list_type   Int
  image_url   String

  list Lists @relation(fields: [list_type], references: [id])

  // Recursive relationship for nested category
  parent   Category?  @relation("CategoryHierarchy", fields: [parent_id], references: [id])
  children Category[] @relation("CategoryHierarchy")

  // Relationship with Intermediate Table (productCategories)
  products ProductCategories[]

  @@map("category")
}

model ProductCategories {
  id Int @id @default(autoincrement())

  product    Product @relation(fields: [product_id], references: [id])
  product_id Int

  category    Category @relation(fields: [category_id], references: [id])
  category_id Int

  // Preventing duplication (each product in a category should only appear once)
  @@unique([product_id, category_id])
  @@map("productcategories")
}

model General {
  id Int @id @default(autoincrement())

  title             String
  desciption        String
  short_description String
  keys              String
  logo_url          String
  address           String
  tel               String
  mobile            String
  email             String
  header_text       String
  about_img_url     String
  about_description String

  @@map("general")
}

model Colors {
  id   Int    @id @default(autoincrement())
  name String

  // Relationship with Intermediate Table (productcolors)
  products ProductColors[]

  // Relationship with Intermediate Table (Images)
  images Images[]

  @@map("colors")
}

model ProductColors {
  id         Int @id @default(autoincrement())
  color_id   Int
  product_id Int

  product Product @relation(fields: [product_id], references: [id])
  color   Colors  @relation(fields: [color_id], references: [id])

  // Preventing duplication (each product in a color pallet should only appear once)
  @@unique([product_id, color_id])
  @@map("productcolors")
}

model Images {
  id        Int    @id @default(autoincrement())
  image_url String
  color_id  Int

  color Colors @relation(fields: [color_id], references: [id])

  // Relationship with Intermediate Table (productimages)
  colors ProductImages[]

  @@map("images")
}

model ProductImages {
  id Int @id @default(autoincrement())

  product_id Int
  image_id   Int

  product Product @relation(fields: [product_id], references: [id])
  image   Images  @relation(fields: [image_id], references: [id])

  @@map("productimages")
}

model Messages {
  id Int @id @default(autoincrement())

  name    String
  email   String
  phone   String
  message String

  @@map("messages")
}
